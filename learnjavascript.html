JavaScript Lessons

Lesson 1: Introduction to JavaScript

JavaScript is a versatile programming language used for adding interactivity and dynamic behavior to websites.

          <script>
            console.log('Hello, world!');
          </script>

Lesson 2: Variables

Learn about variables and how to declare and assign values to them.

          <script>
            var name = 'John';
            var age = 30;
          </script>

Lesson 3: Data Types

Explore different data types in JavaScript such as strings, numbers, booleans, arrays, objects, and more.

          <script>
            var name = 'John'; // string
            var age = 30; // number
            var isStudent = true; // boolean
            var fruits = ['apple', 'banana', 'orange']; // array
            var person = { name: 'John', age: 30 }; // object
          </script>

Lesson 4: Operators

Learn about arithmetic, comparison, logical, and assignment operators in JavaScript.

          <script>
            var x = 10;
            var y = 5;
            
            var sum = x + y;
            var isGreater = x > y;
            var logicalAnd = x > 0 && y < 10;
            
            var result = (x > y) ? 'x is greater' : 'y is greater';
          </script>

Lesson 5: Conditional Statements

Use if...else and switch statements to make decisions in JavaScript.

          <script>
            var age = 30;
            
            if (age >= 18) {
              console.log('You are an adult.');
            } else {
              console.log('You are a minor.');
            }
            
            var day = 'Monday';
            
            switch (day) {
              case 'Monday':
                console.log('Today is Monday.');
                break;
              case 'Tuesday':
                console.log('Today is Tuesday.');
                break;
              default:
                console.log('Today is another day.');
            }
          </script>

Lesson 6: Loops

Use loops to repeat blocks of code in JavaScript.

          <script>
            // For loop
            for (var i = 0; i < 5; i++) {
              console.log(i);
            }
            
            // While loop
            var count = 0;
            while (count < 5) {
              console.log(count);
              count++;
            }
            
            // Do-while loop
            var num = 0;
            do {
              console.log(num);
              num++;
            } while (num < 5);
          </script>

Lesson 7: Functions

Learn how to define and call functions in JavaScript.

          <script>
            function greet(name) {
              return 'Hello, ' + name + '!';
            }
            
            var message = greet('John');
            console.log(message); // Output: 'Hello, John!'
          </script>

Lesson 8: Arrays

Work with arrays and perform common array operations.

          <script>
            var fruits = ['apple', 'banana', 'orange'];
            console.log(fruits[0]); // Output: 'apple'
            console.log(fruits.length); // Output: 3
            
            fruits.push('grape');
            console.log(fruits); // Output: ['apple', 'banana', 'orange', 'grape']
          </script>

Lesson 9: Objects

Understand objects and their properties and methods.

          <script>
            var person = {
              name: 'John',
              age: 30,
              greet: function() {
                console.log('Hello, my name is ' + this.name + '.');
              }
            };
            
            console.log(person.name); // Output: 'John'
            person.greet(); // Output: 'Hello, my name is John.'
          </script>

Lesson 10: Scope

Learn about variable scope in JavaScript.

          <script>
            var x = 10;
            
            function foo() {
              var y = 20;
              console.log(x); // Output: 10
              console.log(y); // Output: 20
            }
            
            foo();
            console.log(x); // Output: 10
            console.log(y); // Error: y is not defined
          </script>

Lesson 11: Objects Constructors

Use constructor functions to create objects.

          <script>
            function Person(name, age) {
              this.name = name;
              this.age = age;
            }
            
            var john = new Person('John', 30);
            console.log(john.name); // Output: 'John'
          </script>

Lesson 12: Closures

Understand closures and their use cases.

          <script>
            function outer() {
              var x = 10;
              
              function inner() {
                console.log(x);
              }
              
              return inner;
            }
            
            var closureFn = outer();
            closureFn(); // Output: 10
          </script>

Lesson 13: DOM Manipulation

Learn how to interact with the Document Object Model (DOM) to modify HTML elements.

          <!-- HTML -->
          <div id="output">Initial content</div>

          <!-- JavaScript -->
          <script>
            var outputElement = document.getElementById('output');
            outputElement.innerHTML = 'Updated content';
          </script>

Lesson 14: Events

Handle events in JavaScript to respond to user interactions.

          <!-- HTML -->
          <button id="myButton">Click me</button>

          <!-- JavaScript -->
          <script>
            var button = document.getElementById('myButton');
            button.addEventListener('click', function() {
              alert('Button clicked!');
            });
          </script>

Lesson 15: Form Handling

Work with form elements and handle form submission in JavaScript.

          <!-- HTML -->
          <form id="myForm">
            <input type="text" name="username" id="usernameInput">
            <button type="submit">Submit</button>
          </form>

          <!-- JavaScript -->
          <script>
            var form = document.getElementById('myForm');
            form.addEventListener('submit', function(event) {
              event.preventDefault();
              var username = document.getElementById('usernameInput').value;
              alert('Hello, ' + username + '!');
            });
          </script>

Lesson 16: Promises

Use promises for asynchronous programming and handling asynchronous tasks.

          <script>
            function fetchData() {
              return new Promise(function(resolve, reject) {
                // Simulate fetching data from server
                setTimeout(function() {
                  var data = 'Data received from server';
                  resolve(data);
                }, 2000);
              });
            }
            
            fetchData().then(function(result) {
              console.log(result); // Output: 'Data received from server'
            });
          </script>

Lesson 17: Fetch API

Learn how to make HTTP requests using the Fetch API.

          <script>
            fetch('https://api.example.com/data')
              .then(response => response.json())
              .then(data => console.log(data))
              .catch(error => console.error(error));
          </script>

Lesson 18: Async/Await

Use async/await for more concise and readable asynchronous code.

<script> async function fetchData() { try { var response = await fetch('https://api.example.com/data'); var data = await response.json(); console.log(data); } catch (error) { console.error(error); } } fetchData(); </script>

Lesson 19: Modules

Organize and modularize your JavaScript code using ES modules.

          <!-- module1.js -->
          <script type="module" src="module1.js"></script>

          <!-- module2.js -->
          <script type="module" src="module2.js"></script>

          <!-- main.js -->
          <script type="module">
            import { greet } from './module1.js';
            import { sum } from './module2.js';

            console.log(greet('John')); // Output: 'Hello, John!'
            console.log(sum(2, 3)); // Output: 5
          </script>

Lesson 20: Error Handling

Handle errors and exceptions in JavaScript.

          <script>
            try {
              // Code that may throw an error
              var result = 10 / 0;
              console.log(result); // This line won't be executed
            } catch (error) {
              console.error('Error:', error.message);
            }
          </script>
